let toastTimer = null;
let unsubscribeLabels = null;
let unsubscribeSettings = null;

function showToast(message) {
  window.clearTimeout(toastTimer);
  elements.toast.textContent = message;
  elements.toast.classList.remove("hidden");
  toastTimer = window.setTimeout(() => elements.toast.classList.add("hidden"), 2800);
}

function showLoginMessage(message) {
  elements.loginMessage.textContent = message;
  elements.loginMessage.classList.remove("hidden");
}

function hideLoginMessage() {
  elements.loginMessage.classList.add("hidden");
}

function escapeHtml(value) {
  return String(value ?? "")
    .replaceAll("&", "&amp;")
    .replaceAll("<", "&lt;")
    .replaceAll(">", "&gt;")
    .replaceAll('"', "&quot;")
    .replaceAll("'", "&#039;");
}

function normalizeLabel(raw, id) {
  return {
    id,
    text: String(raw?.text ?? ""),
    category: String(raw?.category ?? "General").trim() || "General",
    fontFamily: String(raw?.fontFamily ?? "Arial"),
    fontSize: Number(raw?.fontSize ?? 36),
    color: String(raw?.color ?? "#111111"),
    bold: Boolean(raw?.bold),
    italic: Boolean(raw?.italic),
    underline: Boolean(raw?.underline),
    visible: raw?.visible !== false,
    sortOrder: Number(raw?.sortOrder ?? 0)
  };
}

function sortedLabels() {
  return [...state.labels].sort((a, b) => {
    const orderDifference = a.sortOrder - b.sortOrder;
    return orderDifference || a.text.localeCompare(b.text);
  });
}

function populateFonts() {
  elements.labelFont.innerHTML = FONT_OPTIONS
    .map(font => `<option value="${escapeHtml(font)}">${escapeHtml(font)}</option>`)
    .join("");
}

function updateStylePreview() {
  const text = elements.labelText.value.trim() || "Preview text";
  elements.stylePreview.textContent = text;
  elements.stylePreview.style.fontFamily = elements.labelFont.value;
  elements.stylePreview.style.fontSize = `${Math.min(Number(elements.labelFontSize.value || 36), 72)}px`;
  elements.stylePreview.style.color = elements.labelColor.value;
  elements.stylePreview.style.fontWeight = elements.labelBold.checked ? "700" : "400";
  elements.stylePreview.style.fontStyle = elements.labelItalic.checked ? "italic" : "normal";
  elements.stylePreview.style.textDecoration = elements.labelUnderline.checked ? "underline" : "none";
}

function resetLabelForm() {
  elements.labelForm.reset();
  elements.editingId.value = "";
  elements.labelFormTitle.textContent = "Create text option";
  elements.labelCategory.value = "General";
  elements.labelSortOrder.value = "0";
  elements.labelFont.value = "Arial";
  elements.labelFontSize.value = "36";
  elements.labelColor.value = "#111111";
  elements.labelVisible.checked = true;
  elements.cancelEditButton.classList.add("hidden");
  updateStylePreview();
}

function beginEditing(label) {
  elements.editingId.value = label.id;
  elements.labelFormTitle.textContent = "Edit text option";
  elements.labelText.value = label.text;
  elements.labelCategory.value = label.category;
  elements.labelSortOrder.value = String(label.sortOrder);
  elements.labelFont.value = label.fontFamily;
  elements.labelFontSize.value = String(label.fontSize);
  elements.labelColor.value = label.color;
  elements.labelBold.checked = label.bold;
  elements.labelItalic.checked = label.italic;
  elements.labelUnderline.checked = label.underline;
  elements.labelVisible.checked = label.visible;
  elements.cancelEditButton.classList.remove("hidden");
  updateStylePreview();
  window.scrollTo({ top: 0, behavior: "smooth" });
  elements.labelText.focus();
}

function renderLabels() {
  const query = elements.adminSearch.value.trim().toLowerCase();
  const labels = sortedLabels().filter(label =>
    label.text.toLowerCase().includes(query) ||
    label.category.toLowerCase().includes(query)
  );

  elements.optionCount.textContent = `${state.labels.length} option${state.labels.length === 1 ? "" : "s"}`;

  if (!labels.length) {
    elements.adminLabelList.innerHTML = `
      <div class="empty-state">
        ${state.labels.length ? "No options match this search." : "No text options exist yet."}
      </div>
    `;
    return;
  }

  elements.adminLabelList.innerHTML = labels
    .map(label => `
      <article class="admin-item ${label.visible ? "" : "hidden-option"}">
        <div>
          <div class="admin-item-text"
            style="
              font-family:${escapeHtml(label.fontFamily)};
              color:${escapeHtml(label.color)};
              font-weight:${label.bold ? "700" : "400"};
              font-style:${label.italic ? "italic" : "normal"};
              text-decoration:${label.underline ? "underline" : "none"};
            ">
            ${escapeHtml(label.text)}
          </div>
          <div class="admin-item-meta">
            ${escapeHtml(label.category)} · ${escapeHtml(label.fontFamily)} ·
            ${label.fontSize}px · Order ${label.sortOrder} ·
            ${label.visible ? "Visible" : "Hidden"}
          </div>
        </div>

        <div class="admin-item-actions">
          <button class="small-button" type="button" data-action="toggle" data-id="${escapeHtml(label.id)}">
            ${label.visible ? "Hide" : "Show"}
          </button>
          <button class="small-button" type="button" data-action="edit" data-id="${escapeHtml(label.id)}">
            Edit
          </button>
          <button class="small-button danger" type="button" data-action="delete" data-id="${escapeHtml(label.id)}">
            Delete
          </button>
        </div>
      </article>
    `)
    .join("");

  elements.adminLabelList.querySelectorAll("[data-action]").forEach(button => {
    button.addEventListener("click", () => handleListAction(button.dataset.action, button.dataset.id));
  });
}

async function handleListAction(action, id) {
  const label = state.labels.find(item => item.id === id);
  if (!label) return;

  try {
    if (action === "toggle") {
      await update(ref(db, `labelMaker/labels/${id}`), {
        visible: !label.visible,
        updatedAt: Date.now()
      });
      showToast(label.visible ? "Option hidden." : "Option shown.");
    } else if (action === "edit") {
      beginEditing(label);
    } else if (action === "delete") {
      if (!window.confirm(`Delete "${label.text}"?`)) return;
      await remove(ref(db, `labelMaker/labels/${id}`));
      if (elements.editingId.value === id) resetLabelForm();
      showToast("Option deleted.");
    }
  } catch (error) {
    console.error(error);
    showToast("The change could not be saved.");
  }
}

function fillSettingsForm() {
  elements.templatePath.value = state.settings.templatePath;
  elements.labelWidth.value = state.settings.labelWidth;
  elements.labelHeight.value = state.settings.labelHeight;
  elements.canvasWidth.value = state.settings.canvasWidth;
  elements.canvasHeight.value = state.settings.canvasHeight;
}

function subscribeToAdminData() {
  unsubscribeLabels?.();
  unsubscribeSettings?.();

  unsubscribeLabels = onValue(
    ref(db, "labelMaker/labels"),
    snapshot => {
      const raw = snapshot.val() || {};
      state.labels = Object.entries(raw).map(([id, value]) => normalizeLabel(value, id));
      renderLabels();
    },
    error => {
      console.error(error);
      showToast("Could not read label options. Check database rules.");
    }
  );

  unsubscribeSettings = onValue(
    ref(db, "labelMaker/settings"),
    snapshot => {
      state.settings = { ...DEFAULT_SETTINGS, ...(snapshot.val() || {}) };
      fillSettingsForm();
    },
    error => {
      console.error(error);
      showToast("Could not read template settings.");
    }
  );
}

async function handleLogin(event) {
  event.preventDefault();
  hideLoginMessage();

  try {
    await signInWithEmailAndPassword(
      auth,
      elements.loginEmail.value.trim(),
      elements.loginPassword.value
    );
    elements.loginForm.reset();
  } catch (error) {
    console.error(error);
    showLoginMessage("Sign-in failed. Check the email/password and confirm Email/Password is enabled in Firebase Authentication.");
  }
}

async function handlePasswordReset() {
  const email = elements.loginEmail.value.trim();

  if (!email) {
    showLoginMessage("Enter your admin email first.");
    elements.loginEmail.focus();
    return;
  }

  try {
    await sendPasswordResetEmail(auth, email);
    showLoginMessage("A password-reset email was sent.");
  } catch (error) {
    console.error(error);
    showLoginMessage("The reset email could not be sent.");
  }
}

async function handleLabelSubmit(event) {
  event.preventDefault();

  const id = elements.editingId.value;
  const record = {
    text: elements.labelText.value.trim(),
    category: elements.labelCategory.value.trim() || "General",
    sortOrder: Number(elements.labelSortOrder.value || 0),
    fontFamily: elements.labelFont.value,
    fontSize: Number(elements.labelFontSize.value || 36),
    color: elements.labelColor.value,
    bold: elements.labelBold.checked,
    italic: elements.labelItalic.checked,
    underline: elements.labelUnderline.checked,
    visible: elements.labelVisible.checked,
    updatedAt: Date.now()
  };

  if (!record.text) {
    showToast("Enter text for the option.");
    return;
  }

  try {
    if (id) {
      await update(ref(db, `labelMaker/labels/${id}`), record);
      showToast("Option updated.");
    } else {
      record.createdAt = Date.now();
      const newReference = push(ref(db, "labelMaker/labels"));
      await set(newReference, record);
      showToast("Option created.");
    }

    resetLabelForm();
  } catch (error) {
    console.error(error);
    showToast("The option could not be saved.");
  }
}

async function seedStarterLabels() {
  if (state.labels.length && !window.confirm("Add the starter options in addition to the current options?")) {
    return;
  }

  try {
    for (const label of STARTER_LABELS) {
      const newReference = push(ref(db, "labelMaker/labels"));
      await set(newReference, {
        ...label,
        createdAt: Date.now(),
        updatedAt: Date.now()
      });
    }
    showToast("Starter options added.");
  } catch (error) {
    console.error(error);
    showToast("Starter options could not be added.");
  }
}

async function handleSettingsSubmit(event) {
  event.preventDefault();

  const settings = {
    templatePath: elements.templatePath.value.trim() || "94207.avif",
    labelWidth: Number(elements.labelWidth.value),
    labelHeight: Number(elements.labelHeight.value),
    canvasWidth: Number(elements.canvasWidth.value),
    canvasHeight: Number(elements.canvasHeight.value),
    updatedAt: Date.now()
  };

  if (
    settings.labelWidth <= 0 ||
    settings.labelHeight <= 0 ||
    settings.canvasWidth < 300 ||
    settings.canvasHeight < 300
  ) {
    showToast("Enter valid template and print dimensions.");
    return;
  }

  try {
    await set(ref(db, "labelMaker/settings"), settings);
    showToast("Template settings saved.");
  } catch (error) {
    console.error(error);
    showToast("Settings could not be saved.");
  }
}

function attachEvents() {
  elements.loginForm.addEventListener("submit", handleLogin);
  elements.resetPasswordButton.addEventListener("click", handlePasswordReset);
  elements.signOutButton.addEventListener("click", () => signOut(auth));
  elements.labelForm.addEventListener("submit", handleLabelSubmit);
  elements.cancelEditButton.addEventListener("click", resetLabelForm);
  elements.seedButton.addEventListener("click", seedStarterLabels);
  elements.adminSearch.addEventListener("input", renderLabels);
  elements.settingsForm.addEventListener("submit", handleSettingsSubmit);

  [
    elements.labelText,
    elements.labelFont,
    elements.labelFontSize,
    elements.labelColor,
    elements.labelBold,
    elements.labelItalic,
    elements.labelUnderline
  ].forEach(element => {
    element.addEventListener("input", updateStylePreview);
    element.addEventListener("change", updateStylePreview);
  });
}

function start() {
  populateFonts();
  resetLabelForm();
  attachEvents();

  onAuthStateChanged(auth, user => {
    state.currentUser = user;

    if (user) {
      elements.authStatus.textContent = user.email || "Admin signed in";
      elements.authStatus.className = "status-badge online";
      elements.signOutButton.classList.remove("hidden");
      elements.loginPanel.classList.add("hidden");
      elements.adminWorkspace.classList.remove("hidden");
      subscribeToAdminData();
    } else {
      unsubscribeLabels?.();
      unsubscribeSettings?.();
      elements.authStatus.textContent = "Signed out";
      elements.authStatus.className = "status-badge";
      elements.signOutButton.classList.add("hidden");
      elements.loginPanel.classList.remove("hidden");
      elements.adminWorkspace.classList.add("hidden");
    }
  });
}

start();
