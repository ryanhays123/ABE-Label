import { db, ref, onValue } from "./firebase-client.js";

const FONT_OPTIONS = [
  "Arial",
  "Arial Black",
  "Avenir",
  "Baskerville",
  "Courier New",
  "Georgia",
  "Helvetica",
  "Impact",
  "Palatino",
  "Tahoma",
  "Times New Roman",
  "Trebuchet MS",
  "Verdana"
];

const DEFAULT_SETTINGS = {
  templatePath: "94207.avif",
  labelWidth: 4,
  labelHeight: 3,
  canvasWidth: 1000,
  canvasHeight: 750
};

const LABEL_OBJECT_PROPERTIES = [
  "id",
  "sourceLabelId",
  "fontFamily",
  "fontSize",
  "fontWeight",
  "fontStyle",
  "underline",
  "fill",
  "textAlign"
];

const state = {
  labels: [],
  settings: { ...DEFAULT_SETTINGS },
  activeCategory: "All",
  templateImage: null,
  history: [],
  historyIndex: -1,
  restoringHistory: false,
  initializedHistory: false
};

const elements = {
  connectionStatus: document.getElementById("connectionStatus"),
  librarySearch: document.getElementById("librarySearch"),
  categoryFilters: document.getElementById("categoryFilters"),
  labelLibrary: document.getElementById("labelLibrary"),
  canvasStage: document.getElementById("canvasStage"),
  canvasFrame: document.getElementById("canvasFrame"),
  templateMessage: document.getElementById("templateMessage"),
  selectionToolbar: document.getElementById("selectionToolbar"),
  selectedFont: document.getElementById("selectedFont"),
  selectedSize: document.getElementById("selectedSize"),
  selectedColor: document.getElementById("selectedColor"),
  boldButton: document.getElementById("boldButton"),
  italicButton: document.getElementById("italicButton"),
  underlineButton: document.getElementById("underlineButton"),
  duplicateButton: document.getElementById("duplicateButton"),
  forwardButton: document.getElementById("forwardButton"),
  backwardButton: document.getElementById("backwardButton"),
  deleteButton: document.getElementById("deleteButton"),
  undoButton: document.getElementById("undoButton"),
  redoButton: document.getElementById("redoButton"),
  clearButton: document.getElementById("clearButton"),
  downloadButton: document.getElementById("downloadButton"),
  printButton: document.getElementById("printButton"),
  toast: document.getElementById("toast")
};

let toastTimer = null;

function showToast(message) {
  window.clearTimeout(toastTimer);
  elements.toast.textContent = message;
  elements.toast.classList.remove("hidden");
  toastTimer = window.setTimeout(() => elements.toast.classList.add("hidden"), 2800);
}

function escapeHtml(value) {
  return String(value ?? "")
    .replaceAll("&", "&amp;")
    .replaceAll("<", "&lt;")
    .replaceAll(">", "&gt;")
    .replaceAll('"', "&quot;")
    .replaceAll("'", "&#039;");
}

function normalizeLabel(raw, id) {
  return {
    id,
    text: String(raw?.text ?? ""),
    category: String(raw?.category ?? "General").trim() || "General",
    fontFamily: String(raw?.fontFamily ?? "Arial"),
    fontSize: Number(raw?.fontSize ?? 36),
    color: String(raw?.color ?? "#111111"),
    bold: Boolean(raw?.bold),
    italic: Boolean(raw?.italic),
    underline: Boolean(raw?.underline),
    visible: raw?.visible !== false,
    sortOrder: Number(raw?.sortOrder ?? 0)
  };
}

function sortLabels(labels) {
  return [...labels].sort((a, b) => {
    const orderDifference = a.sortOrder - b.sortOrder;
    return orderDifference || a.text.localeCompare(b.text);
  });
}

function populateFonts() {
  elements.selectedFont.innerHTML = FONT_OPTIONS
    .map(font => `<option value="${escapeHtml(font)}">${escapeHtml(font)}</option>`)
    .join("");
}

const canvas = new fabric.Canvas("labelCanvas", {
  preserveObjectStacking: true,
  selection: true,
  backgroundColor: "#ffffff"
});

canvas.on("selection:created", syncSelectionToolbar);
canvas.on("selection:updated", syncSelectionToolbar);
canvas.on("selection:cleared", syncSelectionToolbar);
canvas.on("object:modified", () => {
  syncSelectionToolbar();
  saveHistory();
});
canvas.on("object:added", event => {
  if (state.restoringHistory || event.target === state.templateImage) return;
  saveHistory();
});
canvas.on("object:removed", event => {
  if (state.restoringHistory || event.target === state.templateImage) return;
  saveHistory();
});

function showTemplateMessage(message) {
  elements.templateMessage.textContent = message;
  elements.templateMessage.classList.remove("hidden");
}

function hideTemplateMessage() {
  elements.templateMessage.classList.add("hidden");
}

function resizeCanvas() {
  canvas.setWidth(Number(state.settings.canvasWidth));
  canvas.setHeight(Number(state.settings.canvasHeight));
  elements.canvasFrame.style.width = `${Number(state.settings.canvasWidth)}px`;
  elements.canvasFrame.style.height = `${Number(state.settings.canvasHeight)}px`;
  canvas.calcOffset();
}

function removeTemplate() {
  if (state.templateImage) {
    state.restoringHistory = true;
    canvas.remove(state.templateImage);
    state.restoringHistory = false;
    state.templateImage = null;
  }
}

function loadTemplate() {
  resizeCanvas();
  removeTemplate();
  hideTemplateMessage();

  const templatePath = state.settings.templatePath || "94207.avif";
  const imageElement = new Image();
  imageElement.crossOrigin = "anonymous";

  imageElement.onload = () => {
    const image = new fabric.Image(imageElement, {
      left: 0,
      top: 0,
      originX: "left",
      originY: "top",
      selectable: false,
      evented: false,
      hoverCursor: "default",
      lockMovementX: true,
      lockMovementY: true,
      excludeFromExport: false
    });

    image.scaleX = canvas.getWidth() / imageElement.naturalWidth;
    image.scaleY = canvas.getHeight() / imageElement.naturalHeight;
    state.templateImage = image;

    state.restoringHistory = true;
    canvas.add(image);
    canvas.sendToBack(image);
    state.restoringHistory = false;
    canvas.requestRenderAll();

    if (!state.initializedHistory) {
      resetHistory();
      state.initializedHistory = true;
    }
  };

  imageElement.onerror = () => {
    state.templateImage = null;
    canvas.requestRenderAll();
    showTemplateMessage(
      `The template "${templatePath}" could not be loaded. Put it in the same GitHub folder as index.html, or change its path in Admin.`
    );

    if (!state.initializedHistory) {
      resetHistory();
      state.initializedHistory = true;
    }
  };

  imageElement.src = new URL(templatePath, window.location.href).href;
}

function getTextObjects() {
  return canvas.getObjects().filter(object => object !== state.templateImage);
}

function serializeDesign() {
  return JSON.stringify(
    getTextObjects().map(object => object.toObject(LABEL_OBJECT_PROPERTIES))
  );
}

function resetHistory() {
  state.history = [serializeDesign()];
  state.historyIndex = 0;
  updateHistoryButtons();
}

function saveHistory() {
  if (state.restoringHistory || !state.initializedHistory) return;

  const serialized = serializeDesign();
  if (serialized === state.history[state.historyIndex]) return;

  state.history = state.history.slice(0, state.historyIndex + 1);
  state.history.push(serialized);

  if (state.history.length > 60) {
    state.history.shift();
  } else {
    state.historyIndex += 1;
  }

  updateHistoryButtons();
}

function restoreHistory(index) {
  const serialized = state.history[index];
  if (serialized === undefined) return;

  const data = JSON.parse(serialized);
  state.restoringHistory = true;

  getTextObjects().forEach(object => canvas.remove(object));
  canvas.discardActiveObject();

  fabric.util.enlivenObjects(data, objects => {
    objects.forEach(object => canvas.add(object));
    if (state.templateImage) canvas.sendToBack(state.templateImage);
    state.restoringHistory = false;
    state.historyIndex = index;
    canvas.requestRenderAll();
    syncSelectionToolbar();
    updateHistoryButtons();
  });
}

function updateHistoryButtons() {
  elements.undoButton.disabled = state.historyIndex <= 0;
  elements.redoButton.disabled = state.historyIndex >= state.history.length - 1;
}

function undo() {
  if (state.historyIndex > 0) restoreHistory(state.historyIndex - 1);
}

function redo() {
  if (state.historyIndex < state.history.length - 1) {
    restoreHistory(state.historyIndex + 1);
  }
}

function getActiveText() {
  const object = canvas.getActiveObject();
  return object && object.type === "i-text" ? object : null;
}

function syncSelectionToolbar() {
  const object = getActiveText();

  if (!object) {
    elements.selectionToolbar.classList.add("hidden");
    return;
  }

  elements.selectionToolbar.classList.remove("hidden");
  elements.selectedFont.value = object.fontFamily || "Arial";
  elements.selectedSize.value = Math.round(Number(object.fontSize || 36));
  elements.selectedColor.value = /^#[0-9a-f]{6}$/i.test(String(object.fill))
    ? object.fill
    : "#111111";

  elements.boldButton.classList.toggle(
    "active",
    object.fontWeight === "bold" || Number(object.fontWeight) >= 700
  );
  elements.italicButton.classList.toggle("active", object.fontStyle === "italic");
  elements.underlineButton.classList.toggle("active", Boolean(object.underline));
}

function updateActiveText(properties, save = true) {
  const object = getActiveText();
  if (!object) return;

  object.set(properties);
  object.setCoords();
  canvas.requestRenderAll();
  syncSelectionToolbar();
  if (save) saveHistory();
}

function addLabelToCanvas(label, position = null) {
  const textObject = new fabric.IText(label.text, {
    left: position?.x ?? canvas.getWidth() / 2,
    top: position?.y ?? canvas.getHeight() / 2,
    originX: "center",
    originY: "center",
    fontFamily: label.fontFamily,
    fontSize: label.fontSize,
    fill: label.color,
    fontWeight: label.bold ? "bold" : "normal",
    fontStyle: label.italic ? "italic" : "normal",
    underline: label.underline,
    editable: true,
    lockScalingFlip: true,
    transparentCorners: false,
    cornerStyle: "circle",
    cornerColor: "#2457d6",
    borderColor: "#2457d6",
    padding: 5,
    sourceLabelId: label.id
  });

  canvas.add(textObject);
  canvas.setActiveObject(textObject);
  canvas.requestRenderAll();
  syncSelectionToolbar();
}

function duplicateActive() {
  const object = getActiveText();
  if (!object) {
    showToast("Select text to duplicate.");
    return;
  }

  object.clone(clone => {
    clone.set({
      left: Number(object.left || 0) + 24,
      top: Number(object.top || 0) + 24
    });
    canvas.add(clone);
    canvas.setActiveObject(clone);
    canvas.requestRenderAll();
  }, LABEL_OBJECT_PROPERTIES);
}

function deleteActive() {
  const object = getActiveText();
  if (!object) return;
  canvas.remove(object);
  canvas.discardActiveObject();
  canvas.requestRenderAll();
  syncSelectionToolbar();
}

function clearDesign() {
  if (!getTextObjects().length) return;
  if (!window.confirm("Remove all added text and start over?")) return;

  state.restoringHistory = true;
  getTextObjects().forEach(object => canvas.remove(object));
  state.restoringHistory = false;
  canvas.discardActiveObject();
  canvas.requestRenderAll();
  saveHistory();
  syncSelectionToolbar();
}

function getCategories() {
  return [
    "All",
    ...new Set(
      state.labels
        .filter(label => label.visible)
        .map(label => label.category)
        .sort((a, b) => a.localeCompare(b))
    )
  ];
}

function renderCategories() {
  const categories = getCategories();

  if (!categories.includes(state.activeCategory)) {
    state.activeCategory = "All";
  }

  elements.categoryFilters.innerHTML = categories
    .map(category => `
      <button class="category-chip ${category === state.activeCategory ? "active" : ""}"
              type="button" data-category="${escapeHtml(category)}">
        ${escapeHtml(category)}
      </button>
    `)
    .join("");

  elements.categoryFilters.querySelectorAll("[data-category]").forEach(button => {
    button.addEventListener("click", () => {
      state.activeCategory = button.dataset.category;
      renderCategories();
      renderLibrary();
    });
  });
}

function renderLibrary() {
  const query = elements.librarySearch.value.trim().toLowerCase();

  const visibleLabels = sortLabels(state.labels).filter(label => {
    if (!label.visible) return false;
    if (state.activeCategory !== "All" && label.category !== state.activeCategory) return false;
    return (
      label.text.toLowerCase().includes(query) ||
      label.category.toLowerCase().includes(query)
    );
  });

  if (!visibleLabels.length) {
    elements.labelLibrary.innerHTML = `
      <div class="empty-state">
        No label text matches this search. An admin can add or show options from the Admin page.
      </div>
    `;
    return;
  }

  elements.labelLibrary.innerHTML = visibleLabels
    .map(label => `
      <button class="library-item" type="button" draggable="true"
              data-label-id="${escapeHtml(label.id)}">
        <span class="library-text"
          style="
            font-family:${escapeHtml(label.fontFamily)};
            font-size:${Math.min(label.fontSize, 30)}px;
            color:${escapeHtml(label.color)};
            font-weight:${label.bold ? "700" : "400"};
            font-style:${label.italic ? "italic" : "normal"};
            text-decoration:${label.underline ? "underline" : "none"};
          ">
          ${escapeHtml(label.text)}
        </span>
        <span class="library-meta">${escapeHtml(label.category)}</span>
      </button>
    `)
    .join("");

  elements.labelLibrary.querySelectorAll("[data-label-id]").forEach(button => {
    const labelId = button.dataset.labelId;

    button.addEventListener("click", () => {
      const label = state.labels.find(item => item.id === labelId);
      if (label) addLabelToCanvas(label);
    });

    button.addEventListener("dragstart", event => {
      event.dataTransfer.effectAllowed = "copy";
      event.dataTransfer.setData("text/plain", labelId);
    });
  });
}

function canvasPointFromDrop(event) {
  const rectangle = canvas.upperCanvasEl.getBoundingClientRect();

  return {
    x: (event.clientX - rectangle.left) * (canvas.getWidth() / rectangle.width),
    y: (event.clientY - rectangle.top) * (canvas.getHeight() / rectangle.height)
  };
}

function exportMultiplier() {
  const widthMultiplier = (Number(state.settings.labelWidth) * 300) / canvas.getWidth();
  const heightMultiplier = (Number(state.settings.labelHeight) * 300) / canvas.getHeight();
  return Math.max(1, widthMultiplier, heightMultiplier);
}

function createExportDataUrl() {
  canvas.discardActiveObject();
  canvas.requestRenderAll();

  return canvas.toDataURL({
    format: "png",
    multiplier: exportMultiplier(),
    enableRetinaScaling: true
  });
}

function downloadPng() {
  const anchor = document.createElement("a");
  anchor.href = createExportDataUrl();
  anchor.download = `label-${new Date().toISOString().slice(0, 10)}.png`;
  anchor.click();
}

function printLabel() {
  const printWindow = window.open("", "_blank", "width=900,height=700");
  if (!printWindow) {
    showToast("Allow pop-ups so the print window can open.");
    return;
  }

  const dataUrl = createExportDataUrl();
  const width = Number(state.settings.labelWidth);
  const height = Number(state.settings.labelHeight);

  printWindow.document.write(`<!DOCTYPE html>
    <html>
      <head>
        <title>Print Label</title>
        <style>
          html, body { margin: 0; padding: 0; }
          @page { size: ${width}in ${height}in; margin: 0; }
          img {
            display: block;
            width: ${width}in;
            height: ${height}in;
            object-fit: fill;
          }
        </style>
      </head>
      <body>
        <img src="${dataUrl}" alt="Label">
        <script>
          const image = document.querySelector("img");
          image.addEventListener("load", () => {
            window.focus();
            window.print();
          });
          window.addEventListener("afterprint", () => window.close());
        <\/script>
      </body>
    </html>`);
  printWindow.document.close();
}

function subscribeToData() {
  onValue(
    ref(db, "labelMaker/labels"),
    snapshot => {
      const raw = snapshot.val() || {};
      state.labels = Object.entries(raw).map(([id, value]) => normalizeLabel(value, id));
      renderCategories();
      renderLibrary();
      elements.connectionStatus.textContent = "Live data connected";
      elements.connectionStatus.className = "status-badge online";
    },
    error => {
      console.error(error);
      elements.connectionStatus.textContent = "Database unavailable";
      elements.connectionStatus.className = "status-badge error";
      elements.labelLibrary.innerHTML = `
        <div class="empty-state">
          The label options could not be loaded. Check Firebase Realtime Database rules.
        </div>
      `;
    }
  );

  onValue(
    ref(db, "labelMaker/settings"),
    snapshot => {
      state.settings = { ...DEFAULT_SETTINGS, ...(snapshot.val() || {}) };
      loadTemplate();
    },
    error => {
      console.error(error);
      state.settings = { ...DEFAULT_SETTINGS };
      loadTemplate();
    }
  );
}

function attachEvents() {
  elements.librarySearch.addEventListener("input", renderLibrary);

  elements.canvasStage.addEventListener("dragover", event => {
    event.preventDefault();
    event.dataTransfer.dropEffect = "copy";
    elements.canvasStage.classList.add("drag-active");
  });

  elements.canvasStage.addEventListener("dragleave", event => {
    if (!elements.canvasStage.contains(event.relatedTarget)) {
      elements.canvasStage.classList.remove("drag-active");
    }
  });

  elements.canvasStage.addEventListener("drop", event => {
    event.preventDefault();
    elements.canvasStage.classList.remove("drag-active");

    const labelId = event.dataTransfer.getData("text/plain");
    const label = state.labels.find(item => item.id === labelId);
    if (label) addLabelToCanvas(label, canvasPointFromDrop(event));
  });

  elements.selectedFont.addEventListener("change", () => {
    updateActiveText({ fontFamily: elements.selectedFont.value });
  });

  elements.selectedSize.addEventListener("change", () => {
    updateActiveText({ fontSize: Number(elements.selectedSize.value) });
  });

  elements.selectedColor.addEventListener("input", () => {
    updateActiveText({ fill: elements.selectedColor.value }, false);
  });
  elements.selectedColor.addEventListener("change", saveHistory);

  elements.boldButton.addEventListener("click", () => {
    const object = getActiveText();
    if (!object) return;
    const isBold = object.fontWeight === "bold" || Number(object.fontWeight) >= 700;
    updateActiveText({ fontWeight: isBold ? "normal" : "bold" });
  });

  elements.italicButton.addEventListener("click", () => {
    const object = getActiveText();
    if (!object) return;
    updateActiveText({ fontStyle: object.fontStyle === "italic" ? "normal" : "italic" });
  });

  elements.underlineButton.addEventListener("click", () => {
    const object = getActiveText();
    if (!object) return;
    updateActiveText({ underline: !object.underline });
  });

  elements.duplicateButton.addEventListener("click", duplicateActive);
  elements.deleteButton.addEventListener("click", deleteActive);

  elements.forwardButton.addEventListener("click", () => {
    const object = getActiveText();
    if (!object) return;
    canvas.bringForward(object);
    if (state.templateImage) canvas.sendToBack(state.templateImage);
    canvas.requestRenderAll();
    saveHistory();
  });

  elements.backwardButton.addEventListener("click", () => {
    const object = getActiveText();
    if (!object) return;
    canvas.sendBackwards(object);
    if (state.templateImage) canvas.sendToBack(state.templateImage);
    canvas.requestRenderAll();
    saveHistory();
  });

  elements.undoButton.addEventListener("click", undo);
  elements.redoButton.addEventListener("click", redo);
  elements.clearButton.addEventListener("click", clearDesign);
  elements.downloadButton.addEventListener("click", downloadPng);
  elements.printButton.addEventListener("click", printLabel);

  document.addEventListener("keydown", event => {
    const activeElement = document.activeElement;
    const editingCanvasText = getActiveText()?.isEditing;
    const typingInForm = ["INPUT", "TEXTAREA", "SELECT"].includes(activeElement?.tagName);

    if (editingCanvasText || typingInForm) return;

    const commandKey = event.ctrlKey || event.metaKey;
    const key = event.key.toLowerCase();

    if (commandKey && key === "z") {
      event.preventDefault();
      event.shiftKey ? redo() : undo();
    } else if (commandKey && key === "y") {
      event.preventDefault();
      redo();
    } else if (commandKey && key === "d") {
      event.preventDefault();
      duplicateActive();
    } else if (event.key === "Delete" || event.key === "Backspace") {
      event.preventDefault();
      deleteActive();
    }
  });
}

function start() {
  populateFonts();
  attachEvents();
  resizeCanvas();
  subscribeToData();
}

start();
